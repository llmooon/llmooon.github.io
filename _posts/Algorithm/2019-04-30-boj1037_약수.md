```c++
#include<iostream>
#include<algorithm>
using namespace std;

int main() {
	int n, max_num = 0, min_num = 1e9,num;
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d", &num);
		max_num = max(max_num, num);
		min_num = min(min_num, num);
	}
	printf("%d", max_num*min_num);
}
```
