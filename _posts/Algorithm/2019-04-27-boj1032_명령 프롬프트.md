-  문자열 하나씩 각 index를 비교하여 다르면 ? 출력함.
-   다 같으면 . 출력
-  간단한 구현 문제

```c++
#include<iostream>
#include<string>
using namespace std;
char str[55][55];

int main() {
	int num;
	scanf("%d", &num);
	for (int i = 0; i < num; i++) {
		scanf("%s", str[i]);
	}
	for (int i = 0; str[0][i]; i++) {
		bool flag = 1;
		char c = str[0][i];
		for (int j = 1; j < num && flag; j++) {
			if (c != str[j][i]) flag = 0;
		}
		if (flag) printf("%c", c);
		else printf("?");
	}
}
```
